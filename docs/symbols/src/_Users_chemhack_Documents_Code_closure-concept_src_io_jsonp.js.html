1 //Although closure library has jsonp itself, and the implementation is very robust, but has a large dependency. We will go for our own little dirty jsonp implementation, to save 10KB.
  2 goog.provide('sb.io.Jsonp');
  3 
  4 
  5 /**
  6  * Class for cross-domain jsonp requests.
  7  * @param {string} url
  8  * @param {Object} params
  9  * @param {Function} callback
 10  * @constructor
 11  * @export
 12  */
 13 sb.io.Jsonp = function (url, params, callback) {
 14     this.url = url;
 15     this.callback = callback;
 16     this.internalCallback = this.generateCallback();
 17     // Drop the script on the page
 18     var script = document.createElement('script');
 19     script.setAttribute('type', 'text/javascript');
 20     var query = '';
 21     params = params || {};
 22     var key;
 23     for (key in params) {
 24         if (params.hasOwnProperty(key)) {
 25             query += encodeURIComponent(key) + "=" + encodeURIComponent(params[key]) + "&";
 26         }
 27     }
 28     script.setAttribute('src', url + '?' + query + 'callback=sb.io.Jsonp.' + this.internalCallback);
 29     this.script = document.getElementsByTagName('head')[0].appendChild(script);
 30 };
 31 /**
 32  * Fire a jsonp request
 33  * @param {string} url
 34  * @param {Object} params
 35  * @param {Function} callback
 36  * @export
 37  */
 38 sb.io.Jsonp.call=function(url,params,callback){
 39    new sb.io.Jsonp(url,params,callback);
 40 };
 41 
 42 sb.io.Jsonp.prototype.generateCallback = function () {
 43     var possible = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz';
 44     var randomName = '';
 45 
 46     for (var i = 0; i < 15; i++) {
 47         randomName += possible.charAt(Math.floor(Math.random() * possible.length));
 48     }
 49     var self = this;
 50 
 51     sb.io.Jsonp[randomName] = function (data) {
 52         self.callback(data);
 53 
 54         // Cleanup
 55         delete sb.io.Jsonp[randomName];
 56         self.script.parentNode.removeChild(self.script);
 57     };
 58 
 59     return randomName;
 60 };
