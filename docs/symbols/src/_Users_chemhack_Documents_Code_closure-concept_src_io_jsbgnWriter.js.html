 27 /**
 28  *
 29  * @param {sb.Document} doc
 30  * @return {string}
 31  */
 32 sb.io.JsbgnWriter.prototype.write = function (doc) {
 33     var jsbgn = {};
 34     jsbgn['nodes'] = []; //Always use ['key'] to avoid closure compiler renaming it
 35     jsbgn['edges'] = [];
 36     if (doc.lang()) {
 37         jsbgn['sbgnlang'] = sb.io.JsbgnWriter.sbgnlangMapping_[doc.lang()];
 38     }
 39     goog.array.forEach(doc.nodes(), function (node) {
 40         if(node instanceof sb.Port){
 41             return;
 42         }
 43         if (goog.array.contains([sb.NodeType.UnitOfInformation, sb.NodeType.StateVariable], node.type())) {
 44             return;
 45         }
 46         var nodeObj = {};
 47         nodeObj['id'] = node.id();
 48         nodeObj['sbo'] = sb.sbo.NodeTypeMapping[node.type()];
 49         nodeObj['is_abstract'] = false;
 50         var nodeData = {};
 51         nodeObj['data'] = nodeData;
 52 
 53         if (node.clone()) {
 54             nodeData['clone'] = true;
 55         }
 56         if (node.label()) {
 57             nodeData['label'] = node.label();
 58         }
 59         var bbox = node.attr('box');
 60         if (bbox) {
 61             nodeData['x'] = bbox.x;
 62             nodeData['y'] = bbox.y;
 63             nodeData['width'] = bbox.width;
 64             nodeData['height'] = bbox.height;
 65         }
 66         goog.array.forEach(node.children(), function (subNode) {
 67             if(subNode instanceof sb.Port){
 68                 return;
 69             }
 70             if (subNode.type() == sb.NodeType.UnitOfInformation) {
 71                 if (!nodeData['unitofinformation']) {
 72                     nodeData['unitofinformation'] = [];
 73                 }
 74                 goog.array.insert(nodeData['unitofinformation'], subNode.label() ? subNode.label() : "");
 75             } else if (subNode.type() == sb.NodeType.StateVariable) {
 76                 //TODO: state variable
 77             } else {
 78                 if (!nodeData['subnodes']) {
 79                     nodeData['subnodes'] = [];
 80                 }
 81                 goog.array.insert(nodeData['subnodes'], subNode.id());
 82             }
 83         }, this);
 84         goog.array.insert(jsbgn['nodes'], nodeObj);
 85     }, this);
 86     goog.array.forEach(doc.arcs(), function (arc) {
 87         var arcObj = {};
 88         arcObj['id'] = arc.id();
 89         arcObj['sbo'] = 0; //TODO: do the sbo mapping
 90         arcObj['source'] = arc.source().id();
 91         arcObj['target'] = arc.target().id(); //TODO: deal with state vars, etc
 92         var arcData = {};
 93         arcObj['data'] = arcData;
 94 
 95         goog.array.insert(jsbgn['edges'], arcObj);
 96     }, this);
 97     return goog.json.serialize(jsbgn);
 98 };
