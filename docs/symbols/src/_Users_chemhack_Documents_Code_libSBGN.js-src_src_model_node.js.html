  8 /**
  9  * Class for the nodes. Do not use the constructor, use sb.Document.prototype.createNode instead.
 10  * @see sb.Document#createNode
 11  * @param {!sb.Document} document the document to bind
 12  * @constructor
 13  * @extends sb.model.Element
 14  * @export
 15  */
 16 sb.Node = function (document) {
 17     goog.base(this,document);
 18 
 19 };
 20 
 21 goog.inherits(sb.Node, sb.model.Element);
 22 
 23 /**
 24  * Setter/getter of node type. An error will be thrown if the node type is invalid, see sb.NodeType
 25  * @param {sb.NodeType|string=} opt_type
 26  * @return {sb.NodeType|sb.Node} current node type or sb.Node instance for chaining
 27  * @export
 28  */
 29 sb.Node.prototype.type = function (opt_type) {
 30     if (goog.isDef(opt_type)) {
 31         if (!sb.NodeTypeHelper.isNodeTypeSupported(opt_type)) {
 32             throw new Error('Given node type ' + opt_type + ' is not supported.');
 33         }
 34     }
 35     return /** @type{sb.NodeType|sb.Node}*/this.attr('type', opt_type);
 36 };
 37 
 38 /**
 39  * Setter/getter of node label.
 40  * @param {string=} opt_label label value to set
 41  * @return {string|sb.Node} current label or sb.Node instance for chaining
 42  * @export
 43  */
 44 sb.Node.prototype.label = function (opt_label) {
 45     return /** @type{string|sb.Node}*/this.attr('label', opt_label);
 46 };
 47 
 48 /**

 59 /**
 60  * Create a new node as a sub node of current node.
 61  * @param opt_id Optional id of node.
 62  * @return {sb.Node}
 63  * @export
 64  */
 65 sb.Node.prototype.createSubNode = function (opt_id) {
 66     var node = this.document_.createNode(opt_id);
 67     this.addChild(node);
 68     return node;
 69 };
