1 goog.provide('sb.model.Element');
  2 
  3 goog.require('sb.model.AttributeObject');
  4 
  5 /**
  6  * Base class for sb.Node and sb.Arc.
  7  * @extends sb.model.AttributeObject
  8  * @param {!sb.Document} document the document to bind
  9  * @constructor
 10  * @export
 11  */
 12 sb.model.Element=function(document){
 13     goog.base(this);
 14 
 15     /**
 16      * @protected
 17      * @type {sb.Document}
 18      */
 19     this.document_ = document;
 20 
 21     /**
 22      *
 23      * @type {Array.<sb.model.Element>}
 24      * @private
 25      */
 26     this.children_ = [];
 27 
 28     /**
 29      * The parent node.
 30      * @type {sb.model.Element}
 31      */
 32     this.parent = null;
 33 
 34 };
 35 
 36 goog.inherits(sb.model.Element,sb.model.AttributeObject);
 37 
 38 /**
 39  * Setter/getter of arc id.
 40  * @param {string=} opt_id id value to set
 41  * @return {string|sb.model.Element} current id or sb.model.Element instance for chaining
 42  * @export
 43  */
 44 sb.model.Element.prototype.id = function (opt_id) {
 45     if (goog.isDef(opt_id)) {
 46         this.assertIdUnique_(opt_id);
 47     }
 48     return /** @type{string|sb.model.Element}*/this.attr('id', opt_id, this.document_);
 49 };
 50 
 51 
 52 /**
 53  * Make sure no other element has the same id.
 54  * @param elementId Id to check.
 55  * @private
 56  */
 57 sb.model.Element.prototype.assertIdUnique_ = function (elementId) {
 58     var arc = this.document_.element(elementId);
 59     if (arc && arc != this) {
 60         throw new Error('Given element id ' + elementId + ' already existed');
 61     }
 62 };
 63 /**
 64  * The element to be added as child.
 65  * @param {sb.model.Element} element
 66  * @export
 67  */
 68 sb.model.Element.prototype.addChild = function (element) {
 69     if (element.parent) {
 70         element.parent.removeChild(element);
 71     }
 72     goog.array.insert(this.children_, element);
 73     element.parent = this;
 74 };
 75 
 76 /**
 77  * Remove child element from current element.
 78  * @param {sb.model.Element} element
 79  * @export
 80  */
 81 sb.model.Element.prototype.removeChild = function (element) {
 82     goog.array.remove(this.children_, element);
 83     element.parent = null;
 84 };
 85 
 86 /**
 87  * Return array of child nodes. The array should be treated as read-only. Use sb.model.Element.prototype.addChild and sb.model.Element.prototype.removeChild to modify the child elements.
 88  * @return {Array.<sb.model.Element>}
 89  * @export
 90  */
 91 sb.model.Element.prototype.children = function () {
 92     return this.children_;
 93 };
