 11 /**
 12  * Class for the root structure of a system biology document.
 13  * @param {string=} opt_id Optional id of the document.
 14  * @constructor
 15  * @extends sb.model.AttributeObject
 16  * @export
 17  */
 18 sb.Document = function (opt_id) {
 19     goog.base(this);
 20 
 21     this.id = opt_id;
 22 
 23     /**
 24      * The sequence variable used to name node id.
 25      * @type {number}
 26      * @private
 27      */
 28     this.nodeIdSeq_ = 1;
 29 
 30     /**
 31      * The sequence variable used to name arc id.
 32      * @type {number}
 33      * @private
 34      */
 35     this.arcIdSeq_ = 1;
 36 
 37     /**
 38      * The sequence variable used to name port id.
 39      * @type {number}
 40      * @private
 41      */
 42     this.portIdSeq_ = 1;
 43 
 44     /**
 45      * Id-Object map of all elements.
 46      * @type {goog.structs.Map}
 47      * @private
 48      */
 49     this.elementMap_ = new goog.structs.Map();
 50 
 51     /**
 52      * Node map
 53      * @type {Array.<sb.Node>}
 54      * @private
 55      */
 56     this.nodes_ = [];
 57 
 58     /**
 59      * Arc map
 60      * @type {Array.<sb.Arc>}
 61      * @private
 62      */
 63     this.arcs_ = [];
 64 };
 65 
 66 goog.inherits(sb.Document,sb.model.AttributeObject);
 67 
 68 /**
 69  * Create a new node within the document.
 70  * @param {string=} opt_id Optional id of node.
 71  * @return {sb.Node}
 72  * @export
 73  */
 74 sb.Document.prototype.createNode = function (opt_id) {
 75     var id = opt_id ? opt_id : this.nextNodeId_();
 76     var node = /** @type{sb.Node} */ new sb.Node(this).id(id);
 77     goog.array.insert(this.nodes_,node);
 78     return node;
 79 };
 80 
 81 /**
 82  * Get next unused node id.
 83  * @return {string}
 84  * @private
 85  */
 86 sb.Document.prototype.nextNodeId_ = function () {
 87     var nextNodeId_ = "node" + this.nodeIdSeq_++;
 88     if (this.element(nextNodeId_)) {
 89         return this.nextNodeId_();
 90     } else {
 91         return nextNodeId_;
 92     }
 93 };
 94 
 95 /**
 96  * Readonly. Get all nodes within the document.
 97  * @param {boolean=} opt_noSubNodes If set to true, only nodes without a parent node will be returned.
 98  * @return {Array.<sb.Node>}
 99  * @export
100  */
101 sb.Document.prototype.nodes = function (opt_noSubNodes) {
102     if (opt_noSubNodes) {
103         return goog.array.filter(this.nodes_, function (node, idx, arr) {
104             return node.parent ? false : true;
105         });
106     }
107     return this.nodes_;
108 };
109 
110 /**
111  * Return the node of given id.
112  * @param id
113  * @return {sb.Node}
114  * @export
115  */
116 sb.Document.prototype.node = function (id) {
117     var element = this.element(id);
118     if(element instanceof sb.Node){
119         return element;
120     }
121     return null;
122 };

124 /**

128  */
129 sb.Document.prototype.element = function (id) {
130     return  /** @type{sb.model.Element}*/ this.elementMap_.get(id);

133 /**

146 /**
147  * Create an arc from two nodes
148  * @param {sb.Node} source
149  * @param {sb.Node} target
150  * @return {sb.Arc}
151  * @export
152  */
153 sb.Document.prototype.connect = function (source, target) {
154     var arc = this.createArc();
155     arc.source(source).target(target);
156     return arc;
157 };

197 /**
198  * sb.model.Element will call this method when attribute of their value is changed. This is primiarly used to maintain id-element maps.
199  * @param object{sb.model.Element} The object which attribute is changed.
200  * @param key{string} Attribute name
201  * @param oldValue{string} The old value of attribute
202  * @param newValue{string} The new value of attribute
203  * @protected
204  */
205 sb.Document.prototype.onAttrChange = function (object, key, oldValue, newValue) {
206     if (key == 'id') {
207         if (oldValue) {
208             this.elementMap_.remove(oldValue);
209         }
210         this.elementMap_.set(newValue, object);
211     }
212 };

241 /**
242  * Getter/Setter of SBGN language type
243  * @param {sb.Language=} opt_lang
244  * @return {sb.Language|sb.Document} language type or sb.Document for chaining
245  */
246 sb.Document.prototype.lang=function(opt_lang){
247     if (goog.isDef(opt_lang)) {
248         if (!goog.object.containsValue(sb.Language,opt_lang)) {
249             throw new Error('Given SBGN language type ' + opt_lang + ' is not supported.');
250         }
251     }
252     return /** @type{sb.Language|sb.Document} */ this.attr('language',opt_lang);
253 };
