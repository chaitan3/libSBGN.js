 11 /**
 12  * Base class for xml-based format file reader
 13  * @constructor
 14  */
 15 sb.io.XmlReader = function () {
 16     /**
 17      * ID-Element Map
 18      * @type {goog.structs.Map}
 19      * @private
 20      */
 21     this.idMap_ = null;
 22 };
 23 
 24 /**
 25  *
 26  * @type {goog.debug.Logger}
 27  * @protected
 28  */
 29 sb.io.XmlReader.prototype.logger = goog.debug.Logger.getLogger('sb.io.XmlReader');
 30 
 31 /**
 32  *
 33  * @param {string} xmlText
 34  * @return {Document}
 35  */
 36 sb.io.XmlReader.prototype.parseXmlText = function (xmlText) {
 37     this.logger.info('Parsing xml size:' + xmlText.length);
 38     if(xmlText.charAt(0)=='\ufeff'){
 39         xmlText=xmlText.substr(1,xmlText.length);
 40     }
 41     this.idMap_=new goog.structs.Map();
 42     return goog.dom.xml.loadXml(xmlText);
 43 };
 44 
 45 /**
 46  * Traverse the whole document to build a ID:Element Map
 47  * @param {Element} element
 48  */
 49 sb.io.XmlReader.prototype.buildIdMap = function (element) {
 50     var id = element.getAttribute('id');
 51     if(id){
 52         this.idMap_.set(id,element);
 53     }
 54     goog.array.forEach(element.childNodes, function (child) {
 55         if (child.nodeType == goog.dom.NodeType.ELEMENT) {
 56             this.buildIdMap(child);
 57         }
 58     }, this);
 59 };
 60 
 61 /**
 62  *
 63  * @param {Element} element
 64  */
 65 sb.io.XmlReader.prototype.traverse = function (element) {
 66     this.onElementOpen(element);
 67     goog.array.forEach(element.childNodes, function (child) {
 68         if (child.nodeType == goog.dom.NodeType.ELEMENT) {
 69             this.traverse(child);
 70         }
 71     }, this);
 72     this.onElementClose(element);
 73 };
 74 
 75 /**
 76  * @param {Node} xmlNode
 77  * @protected
 78  */
 79 sb.io.XmlReader.prototype.onElementOpen = goog.abstractMethod;
 80 
 81 /**
 82  * Called on nodeOpen
 83  * @param {Node} xmlNode
 84  * @protected
 85  */
 86 sb.io.XmlReader.prototype.onElementClose = goog.abstractMethod;
