1 goog.provide('sb.util.Stack');
  2 
  3 goog.require('goog.array');
  4 
  5 /**
  6  * A stack data structure
  7  * @constructor
  8  */
  9 sb.util.Stack = function () {
 10     /**
 11      * Internal array for stack implementation
 12      * @type {Array}
 13      * @private
 14      */
 15     this.array_ = [];
 16 };
 17 
 18 /**
 19  * Push an element into stack
 20  * @param {*} element
 21  */
 22 sb.util.Stack.prototype.push = function (element) {
 23     //TODO check goog.array.insert
 24     this.array_.push(element);
 25 };
 26 
 27 /**
 28  * Pop the last element.
 29  * @return {*}
 30  */
 31 sb.util.Stack.prototype.pop = function () {
 32     return this.array_.pop();
 33 };
 34 
 35 /**
 36  * Return the last element, but do not delete it.
 37  * @return {*}
 38  */
 39 sb.util.Stack.prototype.peek = function () {
 40     return goog.array.peek(this.array_);
 41 };
 42 
 43 /**
 44  * Return the underlying array
 45  * @return {Array}
 46  */
 47 sb.util.Stack.prototype.array = function () {
 48     return this.array_;
 49 };
