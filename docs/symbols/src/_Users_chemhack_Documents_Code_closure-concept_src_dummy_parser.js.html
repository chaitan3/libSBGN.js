1 goog.provide('sb.dummy');
  2 
  3 goog.require('goog.array');
  4 goog.require('goog.dom.xml');
  5 
  6 sb.dummy.onload=function(xmlText){
  7     var doc1 = goog.dom.xml.loadXml(xmlText);
  8     var maps = doc1.documentElement.firstElementChild.childNodes;
  9     goog.array.forEach(maps, function (node) {
 10         if (node.tagName == "glyph") {
 11             document.write("Glyph ID:" + node.getAttribute('id') + " Class:" + node.getAttribute("class") + "</br>");
 12         } else if (node.tagName == "arc") {
 13             document.write("Arc Source:" + node.getAttribute('source') + " Target:" + node.getAttribute("target") + " Class:" + node.getAttribute("class") + "</br>");
 14         }
 15     });
 16 };
 17 goog.exportSymbol('sb.dummy.onload', sb.dummy.onload);
