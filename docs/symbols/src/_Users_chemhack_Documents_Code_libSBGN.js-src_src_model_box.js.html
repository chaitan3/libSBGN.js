1 goog.provide('sb.Box');

  3 goog.require('goog.math.Box');

  5 /**
  6  * Class representing a box.
  7  * @param {number} opt_x
  8  * @param {number} opt_y
  9  * @param {number} opt_width
 10  * @param {number} opt_height
 11  * @constructor
 12  * @export
 13  */
 14 sb.Box = function (opt_x, opt_y, opt_width, opt_height) {
 15     //TODO: need to investigate more about closure compiler renaming. maybe too risky without export
 16     /**
 17      * Top left corner, x coordinate
 18      * @type {number}
 19      * @public
 20      */
 21     this.x = goog.isDef(opt_x) ? opt_x : 0;
 22     /**
 23      * Top left corner, y coordinate
 24      * @type {number}
 25      * @public
 26      */
 27     this.y = goog.isDef(opt_y) ? opt_y : 0;
 28     /**
 29      * Width of the box.
 30      * @type {number}
 31      * @public
 32      */
 33     this.width = goog.isDef(opt_width) ? opt_width : 0;
 34     /**
 35      * Height of the box.
 36      * @type {number}
 37      * @public
 38      */
 39     this.height = goog.isDef(opt_height) ? opt_height : 0;
 40 };
 41 
 42 
 43 /**
 44  * Test if this box contains another box.
 45  * @param {sb.Box} anotherBox
 46  * @return {boolean}
 47  */
 48 sb.Box.prototype.contains = function (anotherBox) {
 49     return (anotherBox.x >= this.x) && (anotherBox.y >= this.y) && (this.x + this.width) >= (anotherBox.x + anotherBox.width) && (this.y + this.height) >= (anotherBox.y + anotherBox.height);
 50 };
