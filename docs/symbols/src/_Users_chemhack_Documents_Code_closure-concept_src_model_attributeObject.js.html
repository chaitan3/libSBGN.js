1 goog.provide('sb.model.AttributeObject');
  2 
  3 goog.require('goog.structs.Map');
  4 
  5 /**
  6  * Base class for attribute support
  7  * @constructor
  8  */
  9 sb.model.AttributeObject=function(){
 10 
 11     /**
 12      * Internal attribute map
 13      * @type {goog.structs.Map}
 14      * @private
 15      */
 16     this.attrs_ = new goog.structs.Map();
 17 };
 18 
 19 /**
 20  * Setter/getter of attribute.
 21  * @param {string} key key
 22  * @param {*=} opt_value label value to set
 23  * @param {*=} opt_notifyObject object to notify, the object should implement onAttrChange(object, key, oldValue, newValue) method.
 24  * @return {*} current attributeValue or class instance for chaining
 25  * @export
 26  */
 27 sb.model.AttributeObject.prototype.attr = function (key, opt_value, opt_notifyObject) {
 28     if (goog.isDef(opt_value)) {
 29         var oldValue = this.attrs_.get(key);
 30         this.attrs_.set(key, opt_value);
 31         if(opt_notifyObject){
 32             opt_notifyObject.onAttrChange(this, key, oldValue, opt_value);
 33         }
 34         return this;
 35     } else {
 36         return this.attrs_.get(key);
 37     }
 38 };
