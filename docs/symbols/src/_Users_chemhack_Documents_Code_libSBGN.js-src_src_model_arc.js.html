  9 /**
 10  * Class for the arcs. Do not use the constructor, use sb.Document.prototype.createArc instead.
 11  * @param {!sb.Document} document the document to bind
 12  * @constructor
 13  * @extends sb.model.Element
 14  * @export
 15  */
 16 sb.Arc = function (document) {
 17     goog.base(this, document);
 18 };
 19 
 20 goog.inherits(sb.Arc, sb.model.Element);
 21 
 22 /**
 23  * Setter/getter of node type. An error will be thrown if the node type is invalid, see sb.ArcType
 24  * @param {sb.ArcType|string=} opt_type
 25  * @return {sb.ArcType|sb.Arc} current node type or sb.Node instance for chaining
 26  * @export
 27  */
 28 sb.Arc.prototype.type = function (opt_type) {
 29     if (goog.isDef(opt_type)) {
 30         if (!sb.ArcTypeHelper.isArcTypeSupported(opt_type)) {
 31             throw new Error('Given arc type ' + opt_type + ' is not supported.');
 32         }
 33     }
 34     return /** @type{sb.ArcType|sb.Arc}*/this.attr('type', opt_type);
 35 };
 36 
 37 
 38 /**
 39  * Setter/getter of arc source.
 40  * @param {string|sb.model.Element=} opt_source node or its id
 41  * @return {sb.model.Element|sb.Arc} current source or sb.Arc instance for chaining
 42  * @export
 43  */
 44 
 45 sb.Arc.prototype.source = function (opt_source) {
 46     if (opt_source && goog.isString(opt_source)) {
 47         var element = this.document_.element(opt_source);
 48         if (!element) {
 49             throw new Error('Element ' + opt_source + ' do not exist.');
 50         }
 51         opt_source = element;
 52     }
 53     return /** @type{sb.model.Element|sb.Arc}*/this.attr('source', opt_source);
 54 };

 74 /**
 75  * Create a port as a child of current arc.
 76  * @param opt_id Optional id of arc.
 77  * @return {sb.Port}
 78  * @export
 79  */
 80 //This method is implemented twice in both sb.Arc and sb.Node to avoid circular dependency problem. You will know what the bug is if you move this method to sb.model.Element.
 81 sb.Arc.prototype.createPort = function (opt_id) {
 82     var port = this.document_.createPort(opt_id);
 83     this.addChild(port);
 84     return port;
 85 };
