1 goog.provide('sb.io');
  2 
  3 goog.require('sb.io.SbgnReader');
  4 goog.require('sb.io.Jsonp');
  5 goog.require('sb.io.JsbgnWriter');
  6 goog.require('goog.debug.Logger');
  7 goog.require('goog.json');
  8 
  9 sb.io.logger_ = goog.debug.Logger.getLogger('sb.io');
 10 
 11 /**
 12  * Read a sb.Document from file content
 13  * @param {string} text File content in string.
 14  * @param {string} format currently only 'sbgn' is supported
 15  * @return {sb.Document}
 16  * @export
 17  */
 18 sb.io.read = function (text, format) {
 19     sb.io.logger_.fine("Reading format: " + format);
 20     var reader;
 21     if (format == 'sbgn') {
 22         reader = new sb.io.SbgnReader();
 23         sb.io.logger_.fine("using sb.io.SbgnReader");
 24     } else {
 25         throw new Error("Format " + format + " not supported");
 26     }
 27     sb.io.logger_.fine('Parsing xml size:' + text.length);
 28 
 29     return reader.parseText(text);
 30 };
 31 
 32 /**
 33  * Load a document from a url. NOTE: The data will go through our proxy server.
 34  * @param {string} url URL of file to load. Only http and https is supported. Can be on any server.
 35  * @param {string} format currently only 'sbgn' is supported
 36  * @export
 37  */
 38 sb.io.readUrl = function (url, format, callback_Success) {
 39     var proxyUrl = 'http://chemhack.com/jsonp/ba-simple-proxy.php';
 40     new sb.io.Jsonp(proxyUrl, {'url':url}, function (data) {
 41             sb.io.logger_.fine(goog.json.serialize(data));
 42             if (data['status']['http_code'] == 200) {
 43                 var doc = sb.io.read(data['contents'], format);
 44                 callback_Success(doc);
 45             }
 46         }
 47     );
 48 };

 50 /**
 51  * Write a sb.Document to text string
 52  * @param {sb.Document} doc document to write
 53  * @param {string} format currently only 'jsbgn' is supported
 54  * @return {sb.Document}
 55  * @export
 56  */
 57 sb.io.write=function(doc,format){
 58     var writer;
 59     if (format == 'jsbgn') {
 60         writer = new sb.io.JsbgnWriter();
 61         sb.io.logger_.fine("using sb.io.JsbgnWriter");
 62     } else {
 63         throw new Error("Format " + format + " not supported");
 64     }
 65     return writer.write(doc);
 66 };
