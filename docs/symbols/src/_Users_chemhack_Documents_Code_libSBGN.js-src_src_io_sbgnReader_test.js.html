 51 function dumpDocument(doc) {
 52     var treeConfig = goog.ui.tree.TreeControl.defaultConfig;
 53     var tree = new goog.ui.tree.TreeControl('root', treeConfig);
 54     treeConfig['cleardotPath'] = '../../third-party/closure-library/closure/goog/images/tree/cleardot.gif';
 55     var nodeNode = tree.createNode('');
 56     nodeNode.setText('Nodes:');
 57     dumpElements_(tree, nodeNode, doc.nodes(true));
 58     tree.add(nodeNode);
 59     var arcNode = tree.createNode('');
 60     arcNode.setText('Arcs:');
 61     dumpElements_(tree, arcNode, doc.arcs());
 62     tree.add(arcNode);
 63     tree.render(goog.dom.getElement('dump'));
 64     tree.expandAll();
 65 }
 66 
 67 function dumpElements_(tree, treeNode, nodes) {
 68     goog.array.forEach(nodes, function (element) {
 69         var treeChildNode = tree.createNode('');
 70         if (element instanceof sb.Node) {
 71             treeChildNode.setText(element.id() + " " + element.type());
 72         } else if (element instanceof sb.Arc) {
 73             treeChildNode.setText(element.id() + " " + element.type() + " From: " + element.source().id() + " To: " + element.target().id());
 74         } else if (element instanceof sb.Port) {
 75             treeChildNode.setText(element.id());
 76         }
 77 
 78         treeNode.add(treeChildNode);
 79         dumpElements_(tree, treeChildNode, element.children());
 80     },this);
 81 }
