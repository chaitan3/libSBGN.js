 12 /**
 13  * Reader of sbgn class
 14  * @constructor
 15  * @extends sb.io.XmlReader
 16  * @export

 18 sb.io.SbgnReader = function () {
 19     goog.base(this);
 20     /**
 21      * The stack used to parse xml document.
 22      * @type {sb.util.Stack}
 23      * @private
 24      */
 25     this.objStack_ = null;
 26 
 27     /**
 28      * Array used to hold arcs elements in order to set reference after all glyphs are read in.
 29      * @type {Array.<Element>}
 30      * @private
 31      */
 32     this.delayedArcArray_ = null;
 33 
 34     /**
 35      * The sb.Document to work on.
 36      * @type {sb.Document}
 37      * @private
 38      */
 39     this.document_ = null;
 40 
 41     /**
 42      * Temp array used to store compartments
 43      * @type {Array.<sb.Node>}
 44      * @private
 45      */
 46     this.compartments_ = null;
 47 
 48 };

 54 /**
 55  *
 56  * @param text
 57  * @return {sb.Document}
 58  * @export
 59  */
 60 sb.io.SbgnReader.prototype.parseText = function (text) {
 61     this.logger.info('Parsing xml size:' + text.length);
 62     this.objStack_ = new sb.util.Stack();
 63     this.document_ = new sb.Document();
 64     this.compartments_ = [];
 65     this.delayedArcArray_ = [];
 66     var xmlDocument = this.parseXmlText(text);
 67     goog.asserts.assert(xmlDocument.documentElement);
 68     this.traverse(xmlDocument.documentElement);
 69     goog.asserts.assert(this.objStack_.array().length == 0);
 70     goog.array.forEach(this.delayedArcArray_, function (xmlElement) {
 71         var arc_id = xmlElement.getAttribute('id');
 72         var arc = this.document_.arc(arc_id);
 73         this.logger.finest('arc id: ' + arc_id);
 74         var arc_target = xmlElement.getAttribute('target');
 75         this.logger.finest('arc arc_target: ' + arc_target);
 76         var arc_source = xmlElement.getAttribute('source');
 77         this.logger.finest('arc arc_source: ' + arc_source);
 78         arc.source(arc_source).target(arc_target);
 79     }, this);
 80     return this.document_;
 81 };

 89 /**
 90  * @inheritDoc
 91  * @override
 92  */
 93 sb.io.SbgnReader.prototype.onElementOpen = function (xmlElement) {
 94     var tagName = xmlElement.tagName;
 95     tagName = tagName ? tagName.toLocaleLowerCase() : null;
 96     var nodeId = xmlElement.getAttribute('id');
 97     this.logger.finer('xmlElement open: ' + tagName);
 98     var topElementInStack = this.objStack_.peek();
 99 
100     if (tagName == 'glyph') {
101         this.logger.finest('glyph glyph_id: ' + nodeId);
102         var node = (topElementInStack instanceof sb.Node) ? topElementInStack.createSubNode(nodeId) : this.document_.createNode(nodeId);
103         var glyph_class = xmlElement.getAttribute('class');
104         this.logger.finest('glyph glyph_class: ' + glyph_class);
105         node.type(glyph_class);
106         this.objStack_.push(node);
107         if (glyph_class == 'compartment') {
108             goog.array.insert(this.compartments_, node);
109         }
110     } else if (tagName == 'port') {
111         this.logger.finest('port port_id: ' + nodeId);
112         if (topElementInStack instanceof sb.Node || topElementInStack instanceof sb.Arc) {
113             topElementInStack.createPort(nodeId);
114         }
115     } else if (tagName == 'arc') {
116         this.logger.finest('arc arc_id: ' + nodeId);
117         var arc = this.document_.createArc(nodeId);
118         if(!nodeId){
119             xmlElement.setAttribute('id',arc.id());
120             this.logger.finest('no arc id, arc_id automatically set to: ' + xmlElement.getAttribute('id'));
121         }
122         var arc_class = xmlElement.getAttribute('class');
123         this.logger.finest('arc arc_class: ' + arc_class);
124         arc.type(arc_class);
125         this.objStack_.push(arc);
126         goog.array.insert(this.delayedArcArray_, xmlElement);
127     } else if (tagName == 'label') {
128         topElementInStack.label(xmlElement.getAttribute('text'));
129     } else if (tagName == 'bbox') {
130         var box = new sb.Box(Number(xmlElement.getAttribute('x')), Number(xmlElement.getAttribute('y')), Number(xmlElement.getAttribute('w')), Number(xmlElement.getAttribute('h')));
131         if (xmlElement.parentNode.tagName.toLocaleLowerCase() == 'label') {
132             topElementInStack.attr('label.pos', box);
133         } else {
134             topElementInStack.attr('box', box);
135             if ((topElementInStack instanceof sb.Node) && (topElementInStack.type() != sb.NodeType.Compartment)) {
136                 goog.array.forEach(this.compartments_, function (compartment) {
137                     if (compartment.attr('box').contains(box)) {
138                         compartment.addChild(topElementInStack);
139                     }
140                 }, this);
141             }
142         }
143     } else if (tagName == 'start' || tagName == 'end') {
144         if (topElementInStack instanceof sb.Arc) {
145             topElementInStack.attr(tagName, new sb.Point(Number(xmlElement.getAttribute('x')), Number(xmlElement.getAttribute('y'))));
146         }
147     } else if (tagName == 'map') {
148         var language = xmlElement.getAttribute('language');
149         if (language) {
150             this.document_.lang(language);
151         }
152         this.objStack_.push(this.document_);
153     } else if (tagName == 'entity') {
154         topElementInStack.attr('entity', xmlElement.getAttribute('name'));
155     }
156 };

158 /**
159  * @inheritDoc
160  * @override
161  */
162 sb.io.SbgnReader.prototype.onElementClose = function (xmlElement) {
163     var tagName = xmlElement.tagName;
164     this.logger.finer('node close: ' + tagName);
165     if (tagName == 'glyph' || tagName == 'arc' || tagName == 'map') {
166         this.objStack_.pop();
167     }
168 };
